-- GUI Glow Effect (By ChatGPT)
local ScreenGui = Instance.new("ScreenGui")
local Frame = Instance.new("Frame")
local UICorner = Instance.new("UICorner")
local UIStroke = Instance.new("UIStroke")
local TextLabel = Instance.new("TextLabel")
local TweenService = game:GetService("TweenService")

-- Parent GUI
ScreenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling

-- Frame
Frame.Parent = ScreenGui
Frame.AnchorPoint = Vector2.new(0.5,0.5)
Frame.Position = UDim2.new(0.5,0,0.5,0)
Frame.Size = UDim2.new(0,400,0,200)
Frame.BackgroundColor3 = Color3.fromRGB(20,20,20)
Frame.BackgroundTransparency = 0.2

UICorner.CornerRadius = UDim.new(0,15)
UICorner.Parent = Frame

-- ขอบเรืองแสง
UIStroke.Parent = Frame
UIStroke.Thickness = 4
UIStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
UIStroke.Color = Color3.fromRGB(0, 255, 200)

-- Animation Glow Effect
task.spawn(function()
	while true do
		local tween = TweenService:Create(UIStroke, TweenInfo.new(1.5, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut, -1, true), {
			Color = Color3.fromRGB(0, 255, 150)
		})
		tween:Play()
		tween.Completed:Wait()
	end
end)

-- TextLabel
TextLabel.Parent = Frame
TextLabel.AnchorPoint = Vector2.new(0.5,0.5)
TextLabel.Position = UDim2.new(0.5,0,0.5,0)
TextLabel.Size = UDim2.new(0.9,0,0.4,0)
TextLabel.BackgroundTransparency = 1
TextLabel.Text = "ขณะนี้ระบบกำลังอัพเดท"
TextLabel.Font = Enum.Font.GothamBold
TextLabel.TextSize = 26
TextLabel.TextColor3 = Color3.fromRGB(255,255,255)

-- ทำให้ข้อความมีแสง
local glow = Instance.new("UIStroke")
glow.Parent = TextLabel
glow.Thickness = 2
glow.Color = Color3.fromRGB(0,255,200)
